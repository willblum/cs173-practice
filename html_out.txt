 <script>
  const canvas = document.getElementById("matrix");
  const ctx = canvas.getContext("2d");

  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;

  // Array of C-style code snippets
  const codeSnippets = [
    "int main() {",
    "  printf(\"Hello, world!\\n\");",
    "  return 0;",
    "}",
    "for (int i = 0; i < n; ++i) {",
    "  total += array[i];",
    "}",
    "if (a > b) return a;",
    "while (true) {",
    "  // infinite loop",
    "}",
    "char* str = \"CS @ UIUC\";",
    "struct Node { int data; Node* next; };",
    "switch(option) { case 1: break; }",
    "#include <stdio.h>",
    "void func(int x) {",
    "  x *= 2;",
    "}"
  ];

  const fontSize = 14;
  const columns = Math.floor(canvas.width / fontSize);
  const drops = Array.from({ length: columns }, () => ({
    y: Math.random() * canvas.height / fontSize,
    snippet: codeSnippets[Math.floor(Math.random() * codeSnippets.length)],
    charIndex: 0
  }));

  function drawMatrix() {
    ctx.fillStyle = "rgba(0, 0, 0, 0.05)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    ctx.fillStyle = "#00ff00";
    ctx.font = fontSize + "px monospace";

    drops.forEach((drop, i) => {
      const x = i * fontSize;
      const y = drop.y * fontSize;
      const char = drop.snippet.charAt(drop.charIndex);

      ctx.fillText(char, x, y);

      drop.y += 1;
      drop.charIndex++;

      // When the snippet is done or off-screen, reset
      if (drop.charIndex >= drop.snippet.length || y > canvas.height) {
        drop.y = Math.random() * canvas.height / fontSize;
        drop.charIndex = 0;
        drop.snippet = codeSnippets[Math.floor(Math.random() * codeSnippets.length)];
      }
    });
  }

  setInterval(drawMatrix, 50);

  window.addEventListener("resize", () => {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
  });
</script>
